Variables : To store the value in memory location


var : global scoped

let : blocked scoped

const : blocked scoped

blocked scope is also called local scoped



for example:

        
        //example of var

        // var x = 20; //global
        // {
        //     var x = 30; //global
        //     console.log (`Value of x is ${x}`); //result will be 30
        // }
        // console.log(`Value of x is ${x}`); //result will be 30, because var is global scope
         
    

        
        //example of let

        // let a = 30; //global
        // {
        //     let a = 40; //block 
        //     console.log (a); //result will be 40
        // }
        // console.log(a); //result will be 30 and not 40, because let is block scope
         
    

        
        //example of const

        // const p = 33; //global
        // {
        //     const p = 44; //block 
        //     console.log (p);
        // }
        // console.log(p);
